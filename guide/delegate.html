    {
        KeyPressed?.Invoke(this, e);

    {

            ConsoleKeyInfo keyInfo = Console.ReadKey();
            if (keyInfo.Key == ConsoleKey.Spacebar)
            {
                OnSpaceKeyPressed(new KeyPressEventArgs(keyInfo.Key));
            }

            if (keyInfo.Key == ConsoleKey.Escape)
            {
                break;
            }
        }
    }
}
